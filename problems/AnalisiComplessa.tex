\section{Analisi Complessa}

\Problem{Prodotto infinito del seno}{
Dimostrare che per ogni $z\in\C$ vale la scrittura
\begin{equation*}
	\sin\pi z=\pi z\prod_{n=1}^\infty \left(1-\frac{z^2}{n^2}\right) \punto
\end{equation*}
}
\Hint{
Fate la derivata logaritmica ad entrambi i membri e portate tutto dalla stessa parte. Esce una funzione olomorfa? Limitata?
}
